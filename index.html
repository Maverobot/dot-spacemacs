2.7. Workarounds

2.7.1. Prefer MELPA Stable for shared packages

MELPA occasionally updates archive-contents before all referenced rolling build tarballs are available. Giving MELPA Stable a higher priority lets package.el use stable releases for packages that exist in both archives while still falling back to rolling MELPA for packages that are not released on MELPA Stable. 

(setq configuration-layer-elpa-archives
      '(("melpa-stable" . "stable.melpa.org/packages/")
        ("melpa"        . "melpa.org/packages/")
        ("gnu"          . "elpa.gnu.org/packages/")
        ("nongnu"       . "elpa.nongnu.org/nongnu/")))
(setq package-archive-priorities
      '(("melpa-stable" . 20)
        ("melpa"        . 4)
        ("nongnu"       . 2)
        ("gnu"          . 1)))

2.7.2. Prevent lsp-mode org-element detection from calling Org mode

lsp-mode 10.0.0 probes the org-element API by opening a temporary org-mode buffer during package load. If that org-mode setup errors before lsp-mode finishes loading, Emacs 30 can corrupt cl-generic dispatch: later seq-empty-p calls can lose their list method and reloading lsp-mode can fail with lsp-execute-command is already defined as something else than a generic function.

This config uses Org 9.7+ (currently 9.8.x), so answer the API-version question directly and avoid the temporary org-mode buffer entirely. The advice is installed in user-init before lsp-mode loads; advising after lsp-mode is too late when org-element is already loaded.

Github issue: https://github.com/emacs-lsp/lsp-mode/issues/4898 

(defun my/lsp-org-element-new-api-p ()
  "Return non-nil when Org uses the 9.7+ property-based element API."
  (let ((version (and (fboundp 'org-version)
                      (ignore-errors (org-version)))))
    (if version
        (version<= "9.7" version)
      ;; This Spacemacs config installs a modern Org.  Defaulting to the new
      ;; API keeps lsp-mode from opening an org-mode buffer during startup.
      t)))

(defun my/lsp-detect-org-element-api-without-org-mode (_orig-fn &rest _args)
  "Detect lsp-mode's Org element API support without calling `org-mode'."
  (my/lsp-org-element-new-api-p))

(advice-add 'lsp--detect-org-element-api
            :around #'my/lsp-detect-org-element-api-without-org-mode)

(with-eval-after-load 'lsp-mode
  (setq lsp--org-element-use-new-api (my/lsp-org-element-new-api-p)))

Use idle-time garbage collection instead of a static threshold. Collects garbage when Emacs is idle, keeping GC threshold high during active use. 

(use-package gcmh

3.2.3. Sort Python imports with Ruff

(defvar my/python-ruff-sort-imports-on-save t
  "If non-nil, sort Python imports on save via Ruff's isort rules.")
(put 'my/python-ruff-sort-imports-on-save 'safe-local-variable #'booleanp)

(defun my/python-ruff-sort-imports ()
  "Sort imports in the current Python buffer using Ruff's `I' rules."
  (interactive)
  (when (and my/python-ruff-sort-imports-on-save
             (derived-mode-p 'python-mode 'python-ts-mode))
    (unless (executable-find "ruff")
      (user-error "Cannot sort imports: ruff executable not found"))
    (let ((stdin-filename (or buffer-file-name
                              (expand-file-name "stdin.py" default-directory)))
          (output-buffer (generate-new-buffer " *ruff-import-sort*")))
      (unwind-protect
          (let ((exit-code
                 (call-process-region
                  (point-min) (point-max)
                  "ruff" nil output-buffer nil
                  "check" "--fix-only" "--select" "I"
                  "--stdin-filename" stdin-filename "--silent" "-")))
            (unless (zerop exit-code)
              (user-error "Ruff import sorting failed with exit code %s" exit-code))
            (unless (string= (buffer-string)
                             (with-current-buffer output-buffer (buffer-string)))
              (replace-buffer-contents output-buffer)))
        (kill-buffer output-buffer)))))

(add-hook 'before-save-hook #'my/python-ruff-sort-imports)

3.2.9. Prune deleted LSP workspace folders

  (require 'cl-lib)

  (defun my/lsp-existing-local-or-remote-directory-p (directory)
    "Return non-nil when DIRECTORY should remain in the LSP workspace cache."
    (and (stringp directory)
         (or (file-remote-p directory)
             (file-directory-p directory))))

  (defvar my/lsp--session-for-folder-update nil)
  (defvar my/lsp--folders-for-session-update nil)

  (defun my/lsp-set-session-folders (session folders)
    "Set SESSION's folder list after lsp-mode has defined its struct setter.
Keep the `setf' form quoted so byte-compiling this config before lsp-mode loads
cannot compile against an unknown `lsp-session-folders' setter."
    (let ((my/lsp--session-for-folder-update session)
          (my/lsp--folders-for-session-update folders))
      (eval '(setf (lsp-session-folders my/lsp--session-for-folder-update)
                   my/lsp--folders-for-session-update))))

  ;; Safety: this only mutates `lsp-mode' session metadata and rewrites the
  ;; `.lsp-session-v1' cache file. It never deletes workspace directories/files.
  (defun my/lsp-prune-missing-workspace-folders (&rest _)
    "Remove missing local directories from `lsp-mode' workspace session cache."
    (interactive)
    (when (and (fboundp 'lsp-session)
               (fboundp 'lsp--persist-session))
      (let* ((session (lsp-session))
             (changed nil))
        (cl-labels
            ((keep-directory-p
              (directory)
              (if (my/lsp-existing-local-or-remote-directory-p directory)
                  t
                (setq changed t)
                (message "Pruned missing LSP workspace folder: %s" directory)
                nil))
             (prune-folders
              (folders)
              (let (kept)
                (dolist (directory folders (nreverse kept))
                  (cond
                   ((not (keep-directory-p directory)))
                   ((member directory kept)
                    (setq changed t))
                   (t
                    (push directory kept)))))))
          (my/lsp-set-session-folders
           session (prune-folders (lsp-session-folders session)))
          (let (empty-server-ids)
            (maphash
             (lambda (server-id folders)
               (let ((pruned-folders (prune-folders folders)))
                 (if pruned-folders
                     (puthash server-id pruned-folders
                              (lsp-session-server-id->folders session))
                   (push server-id empty-server-ids))))
             (lsp-session-server-id->folders session))
            (dolist (server-id empty-server-ids)
              (remhash server-id (lsp-session-server-id->folders session))))
          (let (dead-folder-keys)
            (maphash
             (lambda (directory _workspaces)
               (unless (keep-directory-p directory)
                 (push directory dead-folder-keys)))
             (lsp-session-folder->servers session))
            (dolist (directory dead-folder-keys)
              (remhash directory (lsp-session-folder->servers session))))
          (when changed
            (lsp--persist-session session))))))

  (defun my/lsp-prune-missing-workspace-folders-safely (&rest args)
    "Prune stale LSP folders without aborting lsp-mode loading."
    (condition-case err
        (apply #'my/lsp-prune-missing-workspace-folders args)
      (error
       (message "Skipping LSP workspace folder pruning after error: %S" err))))

  (with-eval-after-load 'lsp-mode
    (when (advice-member-p #'my/lsp-prune-missing-workspace-folders 'lsp)
      (advice-remove 'lsp #'my/lsp-prune-missing-workspace-folders))
    (unless (advice-member-p #'my/lsp-prune-missing-workspace-folders-safely 'lsp)
      (advice-add 'lsp :before #'my/lsp-prune-missing-workspace-folders-safely))
    (my/lsp-prune-missing-workspace-folders-safely))

3.2.14. minuet-ai

Use Minuet with an OpenAI-compatible completion endpoint. Run the proxy locally, log in through its dashboard, and export CODEX_PROXY_API_KEY with the dashboard API key.

clang-format does not work properly when editing a file on a remote host or in a docker container with tramp. See the issue here: https://github.com/kljohann/clang-format.el/issues/5

Here upon clang-format-region call, we first check if the file is on a remote host or in a docker container.

  • If not, we call clang-format-region as usual.
  • If yes, we first check if a file with the same path exists on the local disk.
    • If yes, we assume it as the input file name for clang-format-region.
    • If not, we assume the input file is under the $HOME directory.

Depending on where the input file is assumed to be, clang-format will find the .clang-format file for the formatting in a dominant parent directory of the assumed input file path. 

(defun tramp-aware-clang-format (orig-fun start end &optional style assume-file-name)
  (unless assume-file-name
    (setq assume-file-name
          (if (file-remote-p buffer-file-name)
              (let ((maybe-existing-local-buffer-file-name (replace-regexp-in-string "/docker:[^:]+:" "" buffer-file-name)))
                ;; If file `maybe-existing-local-buffer-file-name' exists on local disk, use it.
                (if (file-exists-p maybe-existing-local-buffer-file-name)
                    maybe-existing-local-buffer-file-name
                  ;; Otherwise, use `buffer-file-name' as if it is under the $HOME directory.
                  (concat (getenv "HOME") "/" (file-name-nondirectory buffer-file-name))))
            buffer-file-name)))
  (message "assume-file-name: %s" assume-file-name)
  (apply orig-fun (list start end style assume-file-name)))

(advice-add 'clang-format-region :around #'tramp-aware-clang-format)

\\newunicodechar{≈}{$\\approx$}

    "Trim whitespace from VALUE."
    (replace-regexp-in-string "^[[:space:]]+\\|[[:space:]]+$" "" value))

  (defun my/org-pandoc-eisvogel--font-available-p (font)
    "Return non-nil when FONT is available via `fc-match`."
    (when (executable-find "fc-match")
      (with-temp-buffer
        (when (zerop (call-process "fc-match" nil (current-buffer) nil
                                   "-f" "%{family}\n" font))
          (let* ((families (split-string
                            (replace-regexp-in-string "[\n\r]+" "," (buffer-string))
                            "," t)))
            (member font (mapcar #'my/org-pandoc-eisvogel--trim-whitespace families)))))))

  (defun my/org-pandoc-eisvogel--choose-font (candidates)
    "Return the first available font in CANDIDATES, or the first entry."
    (let ((selected nil)
          (remaining candidates))
      (while (and remaining (not selected))
        (when (my/org-pandoc-eisvogel--font-available-p (car remaining))
          (setq selected (car remaining)))
        (setq remaining (cdr remaining)))
      (or selected (car candidates))))

  (defun my/org-pandoc-eisvogel--ensure-pdf-extension (path)
    "Ensure PATH ends with `.pdf`."
    (if (string-match-p "\\.pdf\\'" path)
        path
      (concat path ".pdf")))

  (defun my/org-pandoc-eisvogel--default-output-path ()
    "Return `<basename>.pdf` for the current Org file."
    (unless (buffer-file-name)
      (user-error "Current Org buffer is not visiting a file"))
    (concat (file-name-sans-extension (buffer-file-name)) ".pdf"))

  (defun my/org-pandoc-eisvogel-export (&optional output-file)
    "Export current Org file to a Pandoc + Eisvogel PDF.

If OUTPUT-FILE is nil, export to `<basename>.pdf` in the current file's
directory."
    (interactive)
    (unless (derived-mode-p 'org-mode)
      (user-error "Not in an Org buffer"))
    (unless (executable-find "pandoc")
      (user-error "Pandoc not found in PATH"))
    (let ((input-file (or (buffer-file-name)
                          (user-error "Current Org buffer is not visiting a file")))
          (output-file (my/org-pandoc-eisvogel--ensure-pdf-extension
                        (or output-file
                            (my/org-pandoc-eisvogel--default-output-path))))
          (main-font (or (getenv "PDF_MAINFONT")
                         (my/org-pandoc-eisvogel--choose-font

          (sans-font (or (getenv "PDF_SANSFONT")

                          '("Noto Sans" "DejaVu Sans"))))
          (mono-font (or (getenv "PDF_MONOFONT")
                         (my/org-pandoc-eisvogel--choose-font
                          '("Fira Code" "Noto Sans Mono" "DejaVu Sans Mono"))))
          (math-font (or (getenv "PDF_MATHFONT")
                         (my/org-pandoc-eisvogel--choose-font
                          '("TeX Gyre Pagella Math" "Latin Modern Math"))))
          (mono-font-options (or (getenv "PDF_MONOFONT_OPTIONS") "Scale=0.88"))

                        output-buffer

                              "--template" "eisvogel"

                              "--variable" (format "monofont=%s" mono-font)
                              "--variable" (format "mathfont=%s" math-font)
                              "--variable" "mainfontoptions=Ligatures=TeX"
                              "--variable" (format "monofontoptions=%s" mono-font-options)
                              "--variable" "code-block-font-size=\\small"))))

3.4.15. org-gcal

Sync Google Calendar events into Org Agenda (read-only fetch). Credentials are stored in ~/.authinfo and never committed to the repo. OAuth tokens are GPG-encrypted via plstore.

Setup instructions (one-time, manual):

  1. Create a Google Cloud project, enable Calendar API, create a Desktop OAuth 2.0 client.
  2. In Google Calendar settings, share any secondary account's calendar with your primary Gmail address.
  3. Generate a dedicated GPG key: gpg --full-generate-key (RSA 4096, no expiry).
  4. Note the key ID: gpg --list-secret-keys --keyid-format=long
  5. Add to ~/.authinfo:
    • machine org-gcal-client-id password <client-id>.apps.googleusercontent.com
    • machine org-gcal-client-secret password <client-secret>
    • machine org-gcal-calendar-id-personal password <personal@gmail.com>
    • machine org-gcal-calendar-id-work password <shared-work-calendar-id>
    • machine org-gcal-gpg-key-id password <GPG-long-key-id>
  6. On first fetch, approve OAuth access in the browser. Tokens are saved and reused automatically.
  (use-package org-gcal
    :defer t
    :init
    ;; Credentials must be set BEFORE org-gcal.el loads, because it calls
    ;; org-gcal-reload-client-id-secret at load time to register the OAuth client.
    (setq org-gcal-client-id     (auth-source-pick-first-password :host "org-gcal-client-id")
          org-gcal-client-secret (auth-source-pick-first-password :host "org-gcal-client-secret"))
    ;; Use asymmetric GPG key for encrypting the OAuth token store.
    ;; Key ID is read from authinfo so it is never hardcoded in this repo.
    (require 'plstore)
    (let ((gpg-key-id (auth-source-pick-first-password :host "org-gcal-gpg-key-id")))
      (when gpg-key-id
        (add-to-list 'plstore-encrypt-to gpg-key-id)))
    ;; Use loopback pinentry so Emacs prompts for GPG passphrases in the minibuffer.
    ;; Requires allow-loopback-pinentry in ~/.gnupg/gpg-agent.conf.
    ;; NOTE: loopback needs an interactive minibuffer; timer-driven syncs cannot
    ;; prompt reliably, so every fetch primes the gpg-agent cache first and aborts
    ;; before org-gcal starts if the OAuth token store cannot be decrypted.
    (setq epg-pinentry-mode 'loopback)
    :config
    ;; Map each calendar ID to its dedicated org file.
    ;; Both files live in ~/org/home/ and are auto-discovered by scan-new-agenda-files.
    (setq org-gcal-fetch-file-alist
          `((,(auth-source-pick-first-password :host "org-gcal-calendar-id-personal")
             . "~/org/home/gcal/gcal-personal.org")
            (,(auth-source-pick-first-password :host "org-gcal-calendar-id-work")
             . "~/org/home/gcal/gcal-work.org")))

     ;; Fetch window: 7 days back, 30 days forward.
    (setq org-gcal-up-days   7
          org-gcal-down-days 30)

    ;; Do not auto-archive old events outside the fetch window.
    (setq org-gcal-auto-archive nil)

    ;; Ask before removing events that appear cancelled/missing on Google's side.
    ;; Avoids silent archiving of items that are simply outside the fetch window.
    (setq org-gcal-update-cancelled-events-with-todo t
          org-gcal-remove-api-cancelled-events       'ask)

    ;; Collect recurring event instances under their parent headline.
    (setq org-gcal-recurring-events-mode 'nested)

    ;; Suppress popup notifications.
    (setq org-gcal-notify-p nil))

  ;; Safe fetch wrapper: primes GPG and clears stale sync lock before fetching.
  ;; Uses org-gcal-fetch (read-only — never pushes local changes back to Google).
  (defun my/org-gcal-fetch-safe ()
    "Fetch Google Calendar events after verifying the OAuth token store decrypts."
    (interactive)
    (let ((interactive-call (called-interactively-p 'interactive)))
      (condition-case err
          (progn
            (my/org-gcal--prime-gpg-cache)
            (require 'org-gcal)
            (when (bound-and-true-p org-gcal--sync-lock)
              (warn "org-gcal: stale sync lock detected, clearing it.")
              (org-gcal--sync-unlock))
            (org-gcal-fetch))
        (error
         (if interactive-call
             (signal (car err) (cdr err))
           (message "org-gcal: fetch skipped: %s"
                    (error-message-string err)))))))

  (defun my/org-gcal--prime-gpg-cache (&optional noerror)
    "Decrypt the org-gcal plstore once to prime the gpg-agent passphrase cache.

Return non-nil when there is no token store yet or decryption succeeds. If
NOERROR is non-nil, report failures as messages and return nil. Otherwise signal
`user-error' so manual and timer-driven fetches stop before org-gcal enters its
deferred OAuth path and reports a later, less useful GPG error."
    (let ((plstore-file (expand-file-name "oauth2-auto.plist" user-emacs-directory))
          store)
      (if (not (file-exists-p plstore-file))
          t
        (condition-case err
            (unwind-protect
                (progn
                  (setq store (plstore-open plstore-file))
                  ;; `plstore-find' consults/decrypts secret entries through the
                  ;; public plstore API, which is stable across Emacs versions.
                  ;; The plaintext values are intentionally discarded.
                  (dolist (entry (plstore-find store nil))
                    (plstore-get store (car entry)))
                  t)
              (when store
                (plstore-close store)))
          (error
           (if noerror
               (progn
                 (message "org-gcal: failed to prime gpg-agent cache: %s"
                          (error-message-string err))
                 nil)
             (user-error
              "org-gcal: cannot decrypt %s; fetch aborted before reading OAuth tokens (%s)"
              (file-name-nondirectory plstore-file)
              (error-message-string err))))))))

  (defun my/org-gcal--prime-gpg-cache-before-sync (&rest _)
    "Prime GPG before any direct `org-gcal-sync' or `org-gcal-fetch' call."
    (my/org-gcal--prime-gpg-cache))

  (with-eval-after-load 'org-gcal
    (unless (advice-member-p #'my/org-gcal--prime-gpg-cache-before-sync
                             'org-gcal-sync)
      (advice-add 'org-gcal-sync :before
                  #'my/org-gcal--prime-gpg-cache-before-sync)))

  ;; Prime once after startup so the first decrypt happens with a minibuffer
  ;; available (loopback pinentry needs it). Run slightly deferred so other
  ;; startup activity doesn't compete for the minibuffer. Fetches still prime
  ;; again and abort safely if this startup attempt fails.
  (add-hook 'emacs-startup-hook
            (lambda ()
              (run-with-idle-timer 2 nil #'my/org-gcal--prime-gpg-cache :noerror)))

  ;; Auto-fetch every 30 minutes. Adjust the interval as needed.
  (run-at-time "30 min" 1800 #'my/org-gcal-fetch-safe)

  ;; Manual sync keybinding: SPC a o s
  (spacemacs/set-leader-keys "aos" 'my/org-gcal-fetch-safe)

3.6.1. Setup instructions (one-time)

mkdir -p ~/.mail/personal/{cur,new,tmp}

# Initial mail sync (after ~/.mbsyncrc is configured)
mbsync -a

# Index mail with mu (after initial sync)
mu init --maildir=~/.mail \
    --my-address=$(pass show or cat ~/.authinfo | grep mu4e-email-personal ...) \
    --my-address=$(...)
